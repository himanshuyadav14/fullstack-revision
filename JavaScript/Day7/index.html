Q3

Convert flat array to tree structure!

const flat = [
  { id: 1, parentId: null, name: "Root" },
  { id: 2, parentId: 1, name: "Child 1" },
  { id: 3, parentId: 1, name: "Child 2" },
  { id: 4, parentId: 2, name: "Grandchild 1" },
];
const buildTree = (flat) => {
  const map = {};
  let root = null;
  flat.forEach(({ id, name }) => {
    map[id] = { id, name, children: [] };
  });
  flat.forEach(({ id, parentId }) => {
    if (parentId === null) root = map[id];
    else map[parentId].children.push(map[id]);
  });
  return root;
};

Q5

Implement limitCalls()!

const limited = limitCalls(fn, 3);
limited(); // works x3, then nothing
const limitCalls = (fn, limit) => {
  let remaining = limit;
  return () => {
    if (remaining > 0) {
      remaining--;
      fn();
    }
  };
};

Q10

Implement createBatcher()!

const createBatcher = (apiFn, delay) => {
  let queue = [];
  let timer = null;
  return function(...args) {
    queue.push(args);
    if(timer) clearTimeout(timer);
    timer = setTimeout(() => {
      queue.forEach(a => apiFn(...a));
      queue = [];
      timer = null;
    }, delay);
  };
};
