Round 2 - Design Patterns

Q4

What is Singleton Pattern? Implement it!

// Singleton = only ONE instance throughout app!
// Use case: database connection, logger, config

const Singleton = (function() {
  let instance;
  function createInstance() {
    return { name: "I am the only instance!" };
  }
  return {
    getInstance: function() {
      if (!instance) instance = createInstance();
      return instance;
    }
  };
})();

const obj1 = Singleton.getInstance();
const obj2 = Singleton.getInstance();
console.log(obj1 === obj2); // true — same instance!
Q5

What is Observer Pattern? Implement EventEmitter!

// Observer = subscription system
// Like YouTube — subscribe, get notified on new video!

class EventEmitter {
  constructor() { this.events = {}; }

  on(event, callback) {
    if (!this.events[event]) this.events[event] = [];
    this.events[event].push(callback);
  }

  emit(event, data) {
    if (this.events[event]) {
      this.events[event].forEach(cb => cb(data));
    }
  }

  off(event, callback) {
    this.events[event] = this.events[event]
      .filter(cb => cb !== callback);
  }
}

const emitter = new EventEmitter();
emitter.on("login", user => console.log(`${user} logged in!`));
emitter.emit("login", "Himanshu"); // "Himanshu logged in!"

Q8

Flatten nested object!

const nested = { a: 1, b: { c: 2, d: { e: 3 } } };
// Output: { a:1, "b.c":2, "b.d.e":3 }
const flattenObject = (obj, prefix = "") => {
  let result = {};
  for (const key in obj) {
    const newKey = prefix ? `${prefix}.${key}` : key;
    if (typeof obj[key] === "object" && obj[key] !== null) {
      Object.assign(result, flattenObject(obj[key], newKey));
    } else {
      result[newKey] = obj[key];
    }
  }
  return result;
};
Q9

Implement leading debounce!

// Executes IMMEDIATELY then ignores for delay time!
debouncedFn(); // executes! ✅
debouncedFn(); // ignored ❌
debouncedFn(); // ignored ❌
// 500ms later...
debouncedFn(); // executes again! ✅
const debounceLeading = (fn, delay) => {
  let timer;
  return function () {
    if (!timer) fn(); // execute if no timer!
    clearTimeout(timer);
    timer = setTimeout(() => {
      timer = null; // reset after delay
    }, delay);
  };
};

Implement retry failed promise N times!

const retry = async (fn, limit) => {
  for (let i = 0; i < limit; i++) {
    try {
      const data = await fn();
      return data; // success!
    } catch (error) {
      console.log(`Attempt ${i + 1} failed`);
      if (i === limit - 1) throw new Error("All attempts failed");
    }
  }
  throw new Error("Failed to fetch data");
};
