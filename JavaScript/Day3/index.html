Q2

Implement debounce from scratch!

function debounce(fn, delay) {
  let timer;
  return function (...args) {
    clearTimeout(timer); // cancel previous timer
    timer = setTimeout(() => fn(...args), delay);
  };
}

const search = debounce((query) => {
  console.log("API call:", query);
}, 500);

search("s");
search("sc");
search("school"); // only this executes after 500ms!

Q4

Implement throttle from scratch!

function throttle(fn, delay) {
  let lastCall = 0;
  return function (...args) {
    const now = Date.now();
    if (now - lastCall >= delay) {
      lastCall = now;
      fn.apply(this, args);
    }
  };
}

Q7

What is Memoization? Implement it!

// Memoization = cache previous results to avoid redundant work!

function memoize(fn) {
  const cache = {};
  return function(...args) {
    const key = JSON.stringify(args);
    if (key in cache) return cache[key]; // from cache!
    cache[key] = fn(...args);
    return cache[key];
  }
}

const memoizedCalc = memoize((n) => n * n);
memoizedCalc(5); // calculates
memoizedCalc(5); // from cache! ✅

Q21

Deep equal two objects!

const deepEqual = (obj1, obj2) => {
  if (obj1 === obj2) return true;
  if (typeof obj1 !== "object" || typeof obj2 !== "object" ||
      obj1 === null || obj2 === null) return false;
  const keys1 = Object.keys(obj1);
  const keys2 = Object.keys(obj2);
  if (keys1.length !== keys2.length) return false;
  for (const key of keys1) {
    if (!deepEqual(obj1[key], obj2[key])) return false;
  }
  return true;
};
Q22

Implement Promise.all() from scratch!

const myPromiseAll = (promises) => {
  return new Promise((resolve, reject) => {
    const results = [];
    let resolved = 0;
    if (promises.length === 0) return resolve([]);
    for (let i = 0; i < promises.length; i++) {
      Promise.resolve(promises[i])
        .then((value) => {
          results[i] = value;
          resolved++;
          if (resolved === promises.length) resolve(results);
        })
        .catch((err) => reject(err));
    }
  });
};
