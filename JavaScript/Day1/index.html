Round 2 - Array Methods

Q8

Difference between slice and splice?

const arr = [1, 2, 3, 4, 5];

// slice(start, end) — does NOT modify original
arr.slice(1, 3); // [2, 3] — end is exclusive
console.log(arr); // [1,2,3,4,5] unchanged!

// splice(start, deleteCount) — MODIFIES original
arr.splice(1, 2); // [2, 3] — removes from arr
console.log(arr); // [1, 4, 5] modified!

// splice can also ADD!
arr.splice(1, 0, 99); // adds 99 at index 1

Q16

What is callback hell? Show example + Promise solution!

// Callback Hell — pyramid of doom!
loginUser("himanshu@gmail.com", function(userId) {
  getCart(userId, function(cart) {
    createOrder(cart, function(orderId) {
      makePayment(orderId, function(paymentId) {
        console.log("Payment done!");
      });
    });
  });
});

// Promise solution — clean chain!
loginUser("himanshu@gmail.com")
  .then(userId => getCart(userId))
  .then(cart => createOrder(cart))
  .then(orderId => makePayment(orderId))
  .catch(err => console.log(err));
Q17

What is Async/Await? Rewrite above using async/await!

async function orderFlow() {
  try {
    const userId = await loginUser("himanshu@gmail.com");
    const cart = await getCart(userId);
    const orderId = await createOrder(cart);
    const paymentId = await makePayment(orderId);
    console.log("Payment done!", paymentId);
  } catch(err) {
    console.log(err);
  }
}
// await pauses function until Promise resolves
// async function always returns a Promise!

Q24

Difference between for...of and for...in?

const arr = [10, 20, 30];
const obj = { a: 1, b: 2, c: 3 };

// for...of — VALUES (arrays/strings)
for(const val of arr) console.log(val); // 10,20,30

// for...in — KEYS (objects)
for(const key in obj) console.log(key); // a,b,c

// for...in on array gives INDEXES!
for(const i in arr) console.log(i); // 0,1,2 — not values!
