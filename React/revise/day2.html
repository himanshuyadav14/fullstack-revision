Q5 What is useEffect? What counts as a side effect? Why not in component body? ▾
9.5/10

  window.addEventListener('resize', handler)
  return () => window.removeEventListener('resize', handler)
}, [])                                                         // Event listener

  const id = setInterval(() => {}, 1000)

// React phases:
// Render phase  → pure calculation, may run multiple times
// Commit phase  → React updates real DOM
// useEffect     → runs after commit, safe for side effects

useEffect(() => {

Q5 useLayoutEffect vs useEffect — difference and when to use? ▾
7/10
⚠️ Got Basic Idea But Missed Key Details
// Exact timing:
// useEffect:       Render → DOM update → Browser PAINTS → useEffect runs
// useLayoutEffect: Render → DOM update → useLayoutEffect runs → Browser PAINTS

// WHY it matters — preventing flicker:
function Tooltip() {
  const [pos, setPos] = useState({ top: 0 })
  const ref = useRef()

  // ❌ useEffect — browser paints FIRST at wrong position → flicker
  useEffect(() => {
    const rect = ref.current.getBoundingClientRect()
    setPos({ top: rect.bottom })
  }, [])

  // ✅ useLayoutEffect — position calculated BEFORE paint → no flicker
  useLayoutEffect(() => {
    const rect = ref.current.getBoundingClientRect()
    setPos({ top: rect.bottom })
  }, [])
}

// WARNING: useLayoutEffect is SYNCHRONOUS
// Heavy logic inside it freezes the UI
// Also: throws warning in SSR (Next.js) — no DOM on server

// RULE: Default to useEffect
// Switch to useLayoutEffect ONLY when you see visual flickering

export function useFetch(url) {
  const [data, setData] = useState(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState(null)

  useEffect(() => {
    const controller = new AbortController()

    setData(null)   // ← reset on URL change
    setError(null)
    setLoading(true)

    fetch(url, { signal: controller.signal })
      .then(res => {
        if (!res.ok) throw new Error(`HTTP error: ${res.status}`) // ← res.ok!
        return res.json()
      })
      .then(data => setData(data))
      .catch(err => {
        if (err.name === 'AbortError') return
        setError(err.message)
      })
      .finally(() => setLoading(false))

    return () => controller.abort()
  }, [url])

  return { data, loading, error }
}
Q2 useLocalStorage hook — works exactly like useState but persists ▾

export function useLocalStorage(key, initialValue) {
  const [value, setValue] = useState(() => {
    try {
      const stored = localStorage.getItem(key)
      return stored !== null ? JSON.parse(stored) : initialValue
    } catch {
      return initialValue
    }
  })

  useEffect(() => {
    try {
      if (value === undefined || value === null) {
        localStorage.removeItem(key)
      } else {
        localStorage.setItem(key, JSON.stringify(value))
      }
    } catch (err) {
      console.error(err)
    }
  }, [value, key]) // ← key in deps!

  return [value, setValue]
}
Lazy initializer ✅ | try/catch in both read/write ✅ | null check ✅ | JSON.parse/stringify ✅
Q3 useDebounce hook — the cleanup IS the debounce mechanism ▾
10/10
✅ Perfect — Know WHY it works
export function useDebounce(value, delay) {
  const [debounced, setDebounced] = useState(value)

  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay)
    return () => clearTimeout(timer) // ← THIS is the debounce
  }, [value, delay])

  return debounced
}

// HOW it works:
// User types 'h'   → timer set for 500ms
// User types 'he'  → cleanup cancels timer, new timer set
// User types 'hel' → cleanup cancels timer, new timer set
// User stops       → timer completes → debounced = 'hel'
// clearTimeout on every keystroke = debounce
Q4 usePrevious hook — useEffect runs AFTER return so ref is always 1 render behind ▾
10/10
✅ Perfect — Understand the Timing
export function usePrevious(value) {
  const ref = useRef()
  useEffect(() => {
    ref.current = value // runs AFTER return — so always one render behind
  }, [value])
  return ref.current
}

// Why it works:
// Render 1: ref=undefined → returns undefined → effect → ref=0
// Render 2: ref=0        → returns 0         → effect → ref=1
// Render 3: ref=1        → returns 1         → effect → ref=2
// ref.current is always the PREVIOUS render's value

Q3 useReducer — when to use over useState? Counter example. ▾
7.5/10
⚠️ Reducer must be defined OUTSIDE component 
// ✅ Reducer OUTSIDE component (pure function, no need inside)
function reducer(state, action) {
  switch (action.type) {
    case 'increment': return { count: state.count + 1 }
    case 'decrement': return { count: state.count - 1 }
    case 'reset':     return { count: 0 }
    default:          return state
  }
}

function App() {
  const [state, dispatch] = useReducer(reducer, { count: 0 })
  return (
    <div>
      <p>{state.count}</p>
      <button onClick={() => dispatch({ type: 'increment' })}>+</button>
      <button onClick={() => dispatch({ type: 'reset' })}>Reset</button>
    </div>
  )
}

// Use useReducer when:
// Multiple related state values
// Complex state transitions
// Many different actions modify same state
// Actions can carry payload: dispatch({ type: 'increment', payload: 5 })
