R1 Q3 React.memo — internals, when it helps, when it doesn't ▾
Score: 7.5/10
How it works internally
// Before re-render, React.memo does:
Object.keys(props).every(key => Object.is(prevProps[key], nextProps[key]))
// true → skip re-render (return cached output)
// false → re-render

// ❌ Fails — new references every render
<Child style={{ color: 'red' }} />       // new object
<Child onClick={() => handle()} />  // new function
<Child items={[1,2,3]} />           // new array

// ✅ Works — primitives or stable references
<Child count={5} label="hello" />
<Child onClick={useCallback(() => {}, [])} />

// Context bypasses React.memo!
const Child = React.memo(() => {
  const theme = useContext(ThemeContext) // context change → re-renders regardless
})

// Custom comparator
const Child = React.memo(Component, (prev, next) =>
  prev.user.id === next.user.id  // true=skip, false=re-render
)

R1 Q5 Lazy loading images — why? How? Native vs Intersection Observer ▾
Score: 8.5/10 
// Method 1 — Native (use by default for all below-fold images)
<img src="photo.jpg" loading="lazy" alt="photo" />

// ❌ Don't lazy load above-fold images — hurts LCP
<img src="hero.jpg" loading="lazy" />  // ❌ hero is immediately visible
<img src="hero.jpg" loading="eager" /> // ✅

// Method 2 — Intersection Observer (custom control, fade-in, analytics)
function LazyImage({ src, alt }) {
  const [loaded, setLoaded] = useState(false)
  const imgRef = useRef()

  useEffect(() => {
    const observer = new IntersectionObserver(([entry]) => {
      if (entry.isIntersecting) {
        setLoaded(true)
        observer.disconnect()
      }
    }, { threshold: 0.1 })
    if (imgRef.current) observer.observe(imgRef.current)
    return () => observer.disconnect()
  }, [])

  return <img ref={imgRef} src={loaded ? src : undefined} alt={alt} />
}

R2 Q2 Reconciliation — what is it? Two key assumptions of diffing algorithm? ▾
Score: 8.5/10 
// Reconciliation: compare old vs new Virtual DOM → minimal real DOM updates

// Assumption 1 — different element type = destroy and rebuild
// Old: <div><Counter /></div>
// New: <span><Counter /></span>
// React sees div→span → unmounts Counter (loses state) → remounts fresh

// Assumption 2 — keys identify stable list elements
// Without keys — React diffs by position → updates ALL items
<li key="apple">Apple</li>  // With keys — React reuses DOM node for same key

// Algorithm complexity:
// Naive: O(n³) — too slow
// React: O(n) — achieved via the two assumptions

// Best approach — use your useDebounce hook from Day 2
function Search() {
  const [query, setQuery] = useState('')
  const [results, setResults] = useState([])
  const [loading, setLoading] = useState(false)
  const debouncedQuery = useDebounce(query, 500)

  useEffect(() => {
    if (!debouncedQuery) { setResults([]); return }

    const controller = new AbortController()
    setLoading(true)

    fetch(`/api/search?q=${debouncedQuery}`, { signal: controller.signal })
      .then(res => {
        if (!res.ok) throw new Error(`HTTP error: ${res.status}`)
        return res.json()
      })
      .then(data => setResults(data))
      .catch(err => { if (err.name !== 'AbortError') console.error(err) })
      .finally(() => setLoading(false))

    return () => controller.abort()
  }, [debouncedQuery])

  return (
    <div>
      <input value={query} onChange={e => setQuery(e.target.value)} placeholder="Search..." />
      {loading && <p>Searching...</p>}
      <ul>{results.map(item => <li key={item.id}>{item.name}</li>)}</ul>
    </div>
  )
}

const UserCard = React.memo(({ user, onSelect }) => (  // React.memo

function UserList({ onSelectUser }) {
  const [users, setUsers] = useState([])
  const [search, setSearch] = useState('')
  const debouncedQuery = useDebounce(search, 500)  // debounce

  useEffect(() => {
    const controller = new AbortController()
    fetch('/api/users', { signal: controller.signal })
      .then(res => { if (!res.ok) throw new Error(); return res.json() })
      .then(data => setUsers(data))
      .catch(err => { if (err.name !== 'AbortError') console.error(err) })
    return () => controller.abort()
  }, [])

  const filtered = useMemo(() => {  // useMemo for filter
    const lower = debouncedQuery.toLowerCase()
    return users.filter(u => u.name.toLowerCase().includes(lower))
  }, [users, debouncedQuery])

  const handleSelect = useCallback(user => onSelectUser(user), [onSelectUser])  // useCallback

  return (
    <div>
      <input value={search} onChange={e => setSearch(e.target.value)} />
      <Virtuoso  // Virtuoso for virtualization
        data={filtered}
        itemContent={(i, user) => <UserCard user={user} onSelect={handleSelect} />}
      />
    </div>
  )
}

// ErrorBoundary.jsx — class component (only way)
class ErrorBoundary extends React.Component {
  state = { hasError: false }
  static getDerivedStateFromError() { return { hasError: true } }
  componentDidCatch(error, info) { console.error(error, info) }
  render() {
    if (this.state.hasError) return (
      <div>
        Failed to load.
        <button onClick={() => this.setState({ hasError: false })}>Retry</button>
      </div>
    )
    return this.props.children
  }
}

// SectionWrapper.jsx — reusable
function SectionWrapper({ Component }) {
  return (
    <ErrorBoundary>
      <Suspense fallback={<Skeleton height={250} />}>
        <Component />
      </Suspense>
    </ErrorBoundary>
  )
}

R3 Q3 usePerformance hook — render count, render time, threshold warning ▾
Score: 9/10 
export const usePerformance = (threshold = 10, componentName = 'Component') => {
  const renderCount = useRef(0)
  const startTime = useRef(performance.now())
  const lastRenderTime = useRef(0)

  // Called during render body — records start time
  startTime.current = performance.now()
  renderCount.current += 1

  useEffect(() => {  // runs after every render
    lastRenderTime.current = performance.now() - startTime.current

    if (renderCount.current > threshold) {
      console.warn(
        `⚠️ ${componentName} re-rendered ${renderCount.current} times. ` +
        `Last: ${lastRenderTime.current.toFixed(2)}ms`
      )
    }
  })

  return { renderCount: renderCount.current, lastRenderTime: lastRenderTime.current }
}

// Usage
const { renderCount, lastRenderTime } = usePerformance(5, 'UserList')
Watch out
Typo: threshold not threshhold (double h is wrong)
setLastRenderTime in state causes extra re-renders — use useRef instead

// useDeferredValue — when you have a VALUE you don't control
function Search({ query }) {  // query from props — can't control it
  const deferredQuery = useDeferredValue(query)
  const isStale = query !== deferredQuery

  const results = useMemo(() =>
    heavySearch(items, deferredQuery)
  , [deferredQuery])

  return <div style={{ opacity: isStale ? 0.5 : 1 }}>{results}</div>

// useTransition — when you control the STATE UPDATE
function TabSwitcher() {
  const [tab, setTab] = useState('home')
  const [isPending, startTransition] = useTransition()

  const switchTab = (newTab) => {
    startTransition(() => {
      setTab(newTab)  // marked as non-urgent
    })
  }

  return (
    <button onClick={() => switchTab('profile')}>
      Profile {isPending && '...'}  // isPending only in useTransition
    </button>
  )
}

// Key difference:
// useDeferredValue = "defer this VALUE" (from props/external)
// useTransition    = "this STATE UPDATE is not urgent" (you control it)
// useTransition gives isPending, useDeferredValue does not
// Both are React 18 concurrent features
